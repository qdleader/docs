// 使用次数：可根据「每个用户」「公司」两个维度进行选择
const GPT_SELECT_ARR = [
  {
    label: '每个用户',
    value: 1
  },
  {
    label: '公司',
    value: 2
  }
];
/**
 * 注意
 * 这个map数组 和 数组是一一对应的
 * 增删的时候需要等号两边同时改
 */
const [
  GPT_SELECT_ARR_MAP
] = [
  GPT_SELECT_ARR
].map(status => {
  return status.reduce((pre, cur) => {
    pre[cur.value] = cur.label;
    return pre;
  }, {});
});

export {
  GPT_SELECT_ARR,
  GPT_SELECT_ARR_MAP
};
