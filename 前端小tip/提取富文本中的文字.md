function getSimpleText(html) {
  let re1 = new RegExp("<.+?>","g") //匹配html标签的正则表达式，"g"是搜索匹配多个符合的内容

  let msg = html.replace(re1,'')
            .replace(/&nbsp;/g,"")
            .replace(/&middot;/g,"")
            .replace(/&ldquo;/g,"")
            .replace(/&rdquo;/g,"")
            .replace(/&mdash;/g,"")
  return msg;
}
