let str = '[abc[bcd[def]]]';

function normalStr(str) {
   let str1 =  str.replace(/\]/g,'')
   let arr = str1.split('[')
    arr = arr.filter(item => item)
   let result = {}
   let obj = result
   while(key = arr.shift()) {
       obj.value = key
       if (!arr.length) break;
       obj.children = {};
       obj = obj.children
   }
   return result
 
}
console.log(normalStr(str));
