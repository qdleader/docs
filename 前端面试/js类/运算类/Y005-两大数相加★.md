


# 两个大数相加


```
let a = "9007199254740991";
let b = "1234567899999999999";


function add(a,b) {
  let maxNum = Math.max(a.length,b.length);
  a = a.padStart(maxNum,0)
  b = b.padStart(maxNum,0)

  let g = 0, s = 0, sum = '';
  for(let i = maxNum -1; i >= 0; i --) {
    g = parseInt(a[i]) + parseInt(b[i]) + s
    s = Math.floor(g/10)
    sum = g%10 + sum
  }
  if(s == 1) {
    sum = '1' + sum
  }
  return sum
}

```
