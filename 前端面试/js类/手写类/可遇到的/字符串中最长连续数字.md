      let nums = str.split('');

      for (let i = 0; i < nums.length; i++) {
        if (nums[i] <= nums[i - 1] || Number.isNaN(Number(nums[i]))) {

          if (i - start > result) {
            result = Math.max(i - start, result);

            start1 = start;
            end1 = i
          }
          start = i;
          if (nums.length - start < result) {
            // return result
            break
          };
        }
      }
      return str.slice(start1, end1)
    }
